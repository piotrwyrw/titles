\pgfmathsetmacro{\fCarrier}{3}
\pgfmathsetmacro{\fMod}{2}

\newcommand{\sineWave}[4] {
	\draw[domain=0:1,smooth,variable=\x,draw=#2,samples=200, opacity=#4] plot (
  		{\paperwidth * \x},
		{sin(\x * 360 * \fCarrier + (#3 * \dPhase) + #1) * (1.5 + sin(\x * 360 * \fMod) * 0.5)}
	);
}
